PROCEDURE max(x, y);
BEGIN
    max := x;
    IF y > x THEN max := y;
END;

PROCEDURE PRINT(x);
    WRITELN(x);

BEGIN
    PRINT(max(3, 5));

    i := 0;
    WHILE 1 + 1 = 2 DO
    BEGIN
        i := i + 1;
        IF i = 3 THEN BREAK;
        WRITELN(i);
        IF i = 2 THEN CONTINUE;
        WRITELN(i);
    END;
END;

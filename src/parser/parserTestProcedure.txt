procedure recur(x,y);
begin
    writeln(x, ' ', y);
    if x=0 then
    begin
        writeln('ZERO');
        recur := 0;
    end;
    else if x<0 then
        writeln('SHOULDNT BE HERE');
    else
    begin
        recur(y,x);
        recur := recur(x-1, y);
    end;
end;

    writeln(2*3 + ' is a cool boolean');
end;
